    ```php
        foreach (['phone', 'email', 'gstin', 'pan', 'logo_url', 'default_validity_days', 'default_cgst_rate', 'default_sgst_rate', 'default_igst_rate'] as $key) {
            if (!array_key_exists($key, $fields)) continue;
            $r = match($key) {
                'phone' => validatePhone($fields[$key]),
                'email' => validateEmail($fields[$key]),
                'gstin' => validateGstin($fields[$key]),
                'pan' => validatePan($fields[$key]),
                'logo_url' => validateUrl($fields[$key]),
                'default_validity_days' => validateDaysInt((int)$fields[$key]),
                default => validateRate((float)$fields[$key]),
            };
            if (!$r['ok']) jsonResponse(false, null, "Invalid $key: {$r['error']}", 400);
        }
        $updated = upsertCompany($pdo, $fields);
        jsonResponse(true, ['company' => $updated]);
    }

    jsonResponse(false, null, 'Method not allowed', 405);
    ```

    **File 2: `public_html/api/crm/quotations-v2/settings/templates.php`**

    Routes by query string: `?type=tc|include|exclude|payment`. Methods: GET (any admin), POST/PUT/DELETE (owner/super_admin).

    Structure:
    ```php
    <?php
    declare(strict_types=1);
    // @generated-by gsd-planner — Phase 1 Plan 03
    require_once __DIR__ . '/../../../../api/admin-guard.php';
    require_once __DIR__ . '/../../../../api/helpers.php';
    require_once __DIR__ . '/../lib/settings_db.php';
    require_once __DIR__ . '/../lib/validation.php';

    $method = $_SERVER['REQUEST_METHOD'];
    $pdo = $GLOBALS['pdo'] ?? null;
    if (!$pdo instanceof PDO) {
        jsonResponse(false, null, 'Database connection unavailable', 500);
    }

    $type = $_GET['type'] ?? null;
    if (!in_array($type, ['tc', 'include', 'exclude', 'payment'], true)) {
        jsonResponse(false, null, "Invalid type. Must be one of: tc, include, exclude, payment", 400);
    }

    if ($method === 'GET') {
        $activeOnly = !isset($_GET['include_inactive']);
        $items = listTemplates($pdo, $type, $activeOnly);
        $key = ($type === 'tc' || $type === 'payment') ? 'templates' : 'items';
        jsonResponse(true, ['type' => $type, $key => $items]);
    }

    if ($method === 'POST') {
        requireRole('owner', 'super_admin');
        $input = getInput();
        try {
            $id = createTemplate($pdo, $type, $input);
            $row = getTemplate($pdo, $type, $id);
            jsonResponse(true, $row, null, 201);
        } catch (InvalidArgumentException $e) {
            jsonResponse(false, null, $e->getMessage(), 400);
        } catch (PDOException $e) {
            // Duplicate slug
            if (str_contains($e->getMessage(), 'Duplicate entry')) {
                jsonResponse(false, null, "Slug already exists", 409);
            }
            throw $e;
        }
    }

    if ($method === 'PUT') {
        requireRole('owner', 'super_admin');
        $id = (int)($_GET['id'] ?? 0);
        if ($id < 1) jsonResponse(false, null, 'Missing or invalid id', 400);
        $input = getInput();
        $ok = updateTemplate($pdo, $type, $id, $input);
        if (!$ok) jsonResponse(false, null, 'Template not found or no changes', 404);
        // If is_default=true was just set, run the unset-others logic
        if (!empty($input['is_default'])) {
            setDefaultTemplate($pdo, $type, $id);
        }
        $row = getTemplate($pdo, $type, $id);
        jsonResponse(true, $row);
    }

    if ($method === 'DELETE') {
        requireRole('owner', 'super_admin');
        $id = (int)($_GET['id'] ?? 0);
        if ($id < 1) jsonResponse(false, null, 'Missing or invalid id', 400);
        $ok = softDeleteTemplate($pdo, $type, $id);
        if (!$ok) jsonResponse(false, null, 'Template not found', 404);
        jsonResponse(true, ['id' => $id, 'is_active' => 0, 'message' => 'Template disabled (soft delete)']);
    }

    jsonResponse(false, null, 'Method not allowed', 405);
    ```
